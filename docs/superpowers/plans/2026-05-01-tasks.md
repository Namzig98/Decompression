# Tasks Implementation Plan (C2)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement C2: the task system. Abstract `TaskObject` base + one concrete `HoldUseTaskObject`, per-crew personal task assignment at round start, server-enforced saboteur fake-completion, hold-Use interaction with synced hold-state for cross-client glow visibility, list HUD + hover-outline UI, and the crew win condition (`Match.EndRound(Crew, "all tasks complete")`) hooked into the existing C1 framework.

**Architecture:** Abstract `TaskObject` base with one concrete `HoldUseTaskObject` subclass shipping in C2; future task types subclass and call `MarkComplete()`. `TaskCoordinator` lives at scene level, subscribes to `Match.RoundStarted`/`RoundEnded`, runs the host-side win-check loop. `TaskAssigner` static helper handles per-crew shuffling and assignment. Networking follows the established patterns: `[Sync(SyncFlags.FromHost)]` for persistent state (AssignedConnectionId, IsCompleted), `[Rpc.Host]` for triggers, `[Rpc.Broadcast]` for cross-client visual events with local-clock timestamps (mirrors Group A's Panel pattern).

**Tech Stack:** s&box (C# / Sandbox.* APIs), Razor `PanelComponent` for HUD, `Component.IPressable` for hold-Use interaction. Builds on existing Match (C1), Player (B), Section (A) primitives.

**Spec:** `docs/superpowers/specs/2026-05-01-tasks-design.md`

```csharp
using System;
using System.Linq;
using Sandbox;

namespace Decompression;

public abstract class TaskObject : Component
{
	[Property] public string DisplayName { get; set; } = "";
	[Property] public string LocationLabel { get; set; } = "";

	[Sync( SyncFlags.FromHost )] public Guid AssignedConnectionId { get; set; }
	[Sync( SyncFlags.FromHost )] public bool IsCompleted { get; set; }

	public bool IsAssignedToLocal =>
		AssignedConnectionId != Guid.Empty
		&& AssignedConnectionId == ( Connection.Local?.Id ?? Guid.Empty );

	public bool IsAssignedAndAlive
	{
		get
		{
			if ( AssignedConnectionId == Guid.Empty ) return false;
			var assignee = Game.ActiveScene?
				.GetAllComponents<Player>()
				.FirstOrDefault( p => p.OwnerConnectionId == AssignedConnectionId );
			return assignee is not null && assignee.IsAlive;
		}
	}

	[Rpc.Host]
	protected void MarkComplete()
	{
		if ( !Networking.IsHost ) return;
		if ( IsCompleted ) return;
		IsCompleted = true;
	}

	[Rpc.Host]
	public void Reset()
	{
		if ( !Networking.IsHost ) return;
		AssignedConnectionId = Guid.Empty;
		IsCompleted = false;
		OnReset();
	}

	protected virtual void OnReset() { }
}
```

`Code/Tasks/TaskAssigner.cs`:

```csharp
using System.Collections.Generic;
using System.Linq;
using Sandbox;

namespace Decompression;

public static class TaskAssigner
{
	// Assigns up to `tasksPerCrew` TaskObjects to each non-saboteur alive
	// player. Pre-condition: caller is host. Saboteurs and dead players get
	// no tasks. If pool is too small for everyone to get the full count,
	// each crew gets as many as available (round-robin) and a warning logs.
	public static void Assign(
		IEnumerable<Player> alivePlayers,
		IEnumerable<TaskObject> availableTasks,
		int tasksPerCrew )
	{
		var crew = alivePlayers
			.Where( p => p.IsValid() && p.IsAlive && !p.IsSaboteur )
			.ToList();
		var pool = availableTasks
			.Where( t => t.IsValid() )
			.ToList();

		if ( crew.Count == 0 )
		{
			Log.Warning( "TaskAssigner.Assign: no alive crew, skipping" );
			return;
		}

		// Fisher-Yates shuffle so assignment isn't deterministic round-to-round.
		for ( int i = pool.Count - 1; i > 0; i-- )
		{
			int j = Game.Random.Int( 0, i );
			(pool[i], pool[j]) = (pool[j], pool[i]);
		}

		var totalNeeded = crew.Count * tasksPerCrew;
		if ( pool.Count < totalNeeded )
		{
			Log.Warning( $"TaskAssigner.Assign: only {pool.Count} tasks for {totalNeeded} needed (crew={crew.Count} × tasksPerCrew={tasksPerCrew}). Each crew will get fewer than requested." );
		}

		// Hand out tasks in round-robin order so under-supply is fairly distributed.
		var index = 0;
		for ( int i = 0; i < tasksPerCrew; i++ )
		{
			foreach ( var player in crew )
			{
				if ( index >= pool.Count ) return;
				pool[index].AssignedConnectionId = player.OwnerConnectionId;
				index++;
			}
		}
	}

	public static void ClearAll( IEnumerable<TaskObject> allTasks )
	{
		foreach ( var task in allTasks )
		{
			if ( task.IsValid() ) task.Reset();
		}
	}
}
```

```csharp
public sealed class HoldUseTaskObject : TaskObject, Component.IPressable
{
	[Property] public float HoldDuration { get; set; } = 3f;
	[Property] public ModelRenderer GlowRenderer { get; set; }

	public Guid HoldingConnectionId { get; private set; }
	public float HoldStartTime { get; private set; }

	private Color initialTint;
	private bool initialTintCaptured;
	private Player cachedHolder;

	bool Component.IPressable.Press( Component.IPressable.Event e )
	{
		// Always route to host. Host validates and decides whether the
		// hold actually counts (assigned crew = real, others = fake).
		BeginHold();
		return true;
	}

	void Component.IPressable.Release( Component.IPressable.Event e )
	{
		EndHold();
	}

	[Rpc.Host]
	public void BeginHold()
	{
		if ( HoldingConnectionId != Guid.Empty ) return;
		var caller = Rpc.Caller;
		if ( caller is null ) return;

		cachedHolder = ResolvePlayerByConnectionId( caller.Id );
		BroadcastHoldStart( caller.Id );
	}

	[Rpc.Host]
	public void EndHold()
	{
		var caller = Rpc.Caller;
		if ( caller is null ) return;
		if ( HoldingConnectionId != caller.Id ) return;

		cachedHolder = null;
		BroadcastHoldEnd();
	}

	[Rpc.Broadcast]
	private void BroadcastHoldStart( Guid connectionId )
	{
		HoldingConnectionId = connectionId;
		HoldStartTime = Time.Now;
	}

	[Rpc.Broadcast]
	private void BroadcastHoldEnd()
	{
		HoldingConnectionId = Guid.Empty;
		HoldStartTime = 0f;
	}

	protected override void OnReset()
	{
		if ( Networking.IsHost && HoldingConnectionId != Guid.Empty )
		{
			BroadcastHoldEnd();
		}
	}

	private static Player ResolvePlayerByConnectionId( Guid connectionId )
	{
		return Game.ActiveScene?
			.GetAllComponents<Player>()
			.FirstOrDefault( p => p.OwnerConnectionId == connectionId );
	}

	// Host-side timer + completion validation in Task 6.
	// UpdateGlow in Task 7.
}
```

## Task 6: HoldUseTaskObject host-side completion + validity

**Files:**
- Modify: `Code/Tasks/HoldUseTaskObject.cs`

```csharp
protected override void OnUpdate()
{
	if ( !Networking.IsHost ) return;
	if ( HoldingConnectionId == Guid.Empty ) return;

	// Defense in depth: drop the hold if the holder is no longer connected
	// or no longer alive.
	if ( !IsHolderStillValid() )
	{
		cachedHolder = null;
		BroadcastHoldEnd();
		return;
	}

	// Completion check.
	var elapsed = Time.Now - HoldStartTime;
	if ( elapsed >= HoldDuration )
	{
		var holder = cachedHolder;
		bool shouldComplete =
			holder is not null
			&& holder.IsAlive
			&& !holder.IsSaboteur
			&& AssignedConnectionId == HoldingConnectionId;

		if ( shouldComplete )
		{
			MarkComplete();
		}

		cachedHolder = null;
		BroadcastHoldEnd();
	}
}

private bool IsHolderStillValid()
{
	if ( cachedHolder is null || !cachedHolder.IsValid() )
	{
		cachedHolder = ResolvePlayerByConnectionId( HoldingConnectionId );
	}
	if ( cachedHolder is null ) return false;
	if ( !cachedHolder.IsAlive ) return false;

	return Connection.All.Any( c => c.Id == HoldingConnectionId );
}
```

## Task 10: TaskCoordinator host-side win check

```csharp
protected override void OnUpdate()
{
	if ( !Networking.IsHost ) return;

	var match = Match.Current;
	if ( match is null || match.State != MatchState.Round ) return;

	var scene = Game.ActiveScene;
	if ( scene is null ) return;

	var alivePlayers = scene.GetAllComponents<Player>()
		.Where( p => p.IsAlive )
		.ToList();
	var aliveConnectionIds = alivePlayers
		.Select( p => p.OwnerConnectionId )
		.ToHashSet();

	int totalAssigned = 0;
	int pendingCount = 0;
	foreach ( var task in scene.GetAllComponents<TaskObject>() )
	{
		if ( task.AssignedConnectionId == System.Guid.Empty ) continue;
		if ( !aliveConnectionIds.Contains( task.AssignedConnectionId ) ) continue;
		totalAssigned++;
		if ( !task.IsCompleted ) pendingCount++;
	}

	// Guard: don't crew-win on round start with zero assigned tasks.
	if ( totalAssigned > 0 && pendingCount == 0 )
	{
		match.EndRound( MatchOutcome.Crew, "all tasks complete" );
	}
}
```

```csharp
		.Where( t => t.AssignedConnectionId == connectionId )

		// but we can drive completion via the synced IsCompleted property
		// directly on host (host-only setter).
		task.IsCompleted = true;
	}
	Log.Info( $"decompv2_complete_my_tasks: marked {tasks.Count} task(s) complete for {connectionId}" );
}

[ConCmd( "decompv2_list_tasks" )]
public static void ListTasks()
{
	var scene = Game.ActiveScene;
	if ( scene is null )
	{
		Log.Warning( "decompv2_list_tasks: no active scene" );
		return;
	}

	var tasks = scene.GetAllComponents<TaskObject>().ToList();
	if ( tasks.Count == 0 )
	{
		Log.Info( "decompv2_list_tasks: no TaskObjects in scene" );
		return;
	}

	foreach ( var task in tasks )
	{
		var assigneeName = task.AssignedConnectionId == System.Guid.Empty
			? "(unassigned)"
			: scene.GetAllComponents<Player>()
				.FirstOrDefault( p => p.OwnerConnectionId == task.AssignedConnectionId )
				?.Network.Owner?.DisplayName ?? "(unknown)";
		var status = task.IsCompleted ? "DONE" : "pending";
		Log.Info( $"  [{task.LocationLabel}] {task.DisplayName} → {assigneeName} ({status})" );
	}
}

{

	{
```

```razor
		var match = Decompression.Match.Current;
		if ( match is null || match.State != MatchState.Round )
		{
			showList = false;
			return;
		}

		var localId = Connection.Local?.Id ?? System.Guid.Empty;
		if ( localId == System.Guid.Empty )
		{
			showList = false;
			return;
		}

		var myTasks = Game.ActiveScene?
			.GetAllComponents<Decompression.TaskObject>()
			.Where( t => t.AssignedConnectionId == localId )
			.ToList();

		if ( myTasks is null || myTasks.Count == 0 )
		{
			showList = false;
			return;
		}

		showList = true;
		entries = myTasks
			.Select( t => new TaskEntry( t.LocationLabel, t.DisplayName, t.IsCompleted ) )
			.ToList();
	}

	protected override int BuildHash()
	{
		var hash = entries.Count;
		foreach ( var e in entries )
		{
			hash = System.HashCode.Combine( hash, e.Done );
		}
		return System.HashCode.Combine( hash, showList );
	}
}
```

`Code/Tasks/TaskHoverOutline.razor`:

```razor
@using System.Linq;
@using Sandbox;
@using Sandbox.UI;
@namespace Decompression
@inherits PanelComponent

<root>
	@if ( showOutline )
	{
		<div class="hover-prompt">
			<div class="prompt-text">[E] @hoveredName</div>
		</div>
	}
</root>

@code {
	private bool showOutline;
	private string hoveredName = "";

	protected override void OnUpdate()
	{
		showOutline = false;

		var match = Decompression.Match.Current;
		if ( match is null || match.State != MatchState.Round ) return;

		var camera = Game.ActiveScene?
			.GetAllComponents<CameraComponent>()
			.FirstOrDefault( c => c.IsMainCamera );
		if ( camera is null ) return;

		// Trace forward from camera ~150u (the player's interact reach).
		const float reach = 150f;
		var origin = camera.WorldPosition;
		var direction = camera.WorldRotation.Forward;
		var trace = Game.ActiveScene.Trace
			.Ray( origin, origin + direction * reach )
			.Run();

		if ( !trace.Hit || trace.GameObject is null ) return;

		var task = trace.GameObject.Components.Get<Decompression.TaskObject>( includeDisabled: true )
			?? trace.GameObject.Root?.Components.Get<Decompression.TaskObject>( includeDisabled: true );

		if ( task is null ) return;
		if ( !task.IsAssignedToLocal ) return;
		if ( task.IsCompleted ) return;

		showOutline = true;
		hoveredName = $"{task.LocationLabel} — {task.DisplayName}";
	}

	protected override int BuildHash() => System.HashCode.Combine( showOutline, hoveredName );
}
```
